local ArcadeCore = GameModules["Arcade/Core"]
local MainMap = GameModules["Maps"]
local RoomMod = GameModules["Mods/NewRoomMod"]

local RoomGenerator = RoomMod.RoomGenerator

local ROOM_NAME = "RoomName"
local SPAWN_POINT = { x = 23.62, y = 14 }

function ValidateAndLoadSaveData()
    -- This will probably be done in a different scene
    local maskOptions = {
        color = { r = 0, g = 0, b = 0 },
        duration = 1
    }

    DCEI.ShowScreenMaskWithAlpha(1, maskOptions)

    local savedRoomType = DCEI.Save.Get(ROOM_NAME)
    if savedRoomType ~= nil then
        local roomPath = RoomMod.GetRoomPath(savedRoomType)
        DCEI.PlayLevel(roomPath, savedRoomType)
    else
        RoomGenerator:ClearSaveData() -- If nothing exits reload a fresh game
    end

    DCEI.ShowScreenMaskWithAlpha(0, maskOptions)
end

ValidateAndLoadSaveData()
MainMap.Start(SPAWN_POINT, nil, nil, nil)
